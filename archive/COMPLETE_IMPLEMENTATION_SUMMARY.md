### Layla AI Assistant - Basic Usage

```kotlin
// Initialize Layla AI Assistant (all phases)
val layla = LaylaAssistant()
layla.initialize().getOrThrow()

// Start a conversation
val conversationId = layla.startChat("Hello Layla!")

// Generate an image
val sdService = layla.getStableDiffusionService()

// Process a document

// Get privacy dashboard
val privacyService = layla.getPrivacyService()
val dashboard = privacyService.getPrivacyDashboard()

// Monitor system health
val analyticsService = layla.getAnalyticsService()
val health = analyticsService.getHealthDashboard()
```

### Cognitive Computing - Meta-Cognition Usage

```kotlin
// Initialize cognitive components
val cognitiveEngine = CognitiveEngine()

// Phase 5: Perform meta-cognitive introspection
val metaSystem = cognitiveEngine.getMetaCognitiveSystem()
val introspection = metaSystem.performIntrospection()
println("Self-awareness level: ${introspection.selfAwarenessLevel}")

// Extract cognitive rules
val rules = metaSystem.extractCognitiveRules(minConfidence = 0.75)
println("Extracted ${rules.size} cognitive rules")

// Evolutionary optimization
val optimizer = cognitiveEngine.getEvolutionaryOptimizer()
val optimized = optimizer.optimize(generations = 50)
println("Best fitness: ${optimized.bestFitness}")

// Recursive verification
val verifier = cognitiveEngine.getRecursiveVerifier()
val verification = verifier.performRecursiveVerification(depth = 3)
println("Verification confidence: ${verification.overallConfidence}")

// Phase 6: Monitor unified cognitive state
val monitor = cognitiveEngine.getUnifiedMonitor()
val state = monitor.getCurrentState()
println("System health: ${state.healthMetrics.overallHealth}")

// Verify system consistency
val consistencyVerifier = cognitiveEngine.getConsistencyVerifier()
val consistency = consistencyVerifier.verifySystemConsistency()
```
